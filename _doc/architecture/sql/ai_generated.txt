```sql
SELECT
  p.product_name,
  SUM(oi.quantity) AS total_quantity_sold
FROM products AS p
JOIN order_items AS oi
  ON p.id = oi.product_id
JOIN orders AS o
  ON oi.order_id = o.id
WHERE
  EXTRACT(MONTH FROM o.order_date) = 7
GROUP BY
  p.product_name
ORDER BY
  total_quantity_sold DESC
LIMIT 5;
```
